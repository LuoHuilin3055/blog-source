# 拼接GIF每一帧
```python
from PIL import Image
import numpy as np

img = Image.open("flag.gif")

frames = []
try:
    i = 0
    while True:
        img.seek(i)
        frames.append(img.convert("RGB"))
        i += 1
except EOFError:
    pass

arrays = [np.array(f) for f in frames]

h = arrays[0].shape[0]
w_total = sum(a.shape[1] for a in arrays)

canvas = np.zeros((h, w_total, 3), dtype=np.uint8)

x = 0
for a in arrays:
    canvas[:, x:x+a.shape[1]] = a
    x += a.shape[1]

Image.fromarray(canvas).save("flag.png")
```
